\begin{align}
    G_n = \frac{E_n - E_{n-1}}{a} - q \psi_n\dagger \psi_n - \frac{(-1)^n - 1}{2} \: ,
\end{align}
\begin{align} \label{eq:GeneratorU(1)Symmetry}
    G_n = - S_n^z + S_{n-1}^z - \frac{q}{2} \big( - \sigma_n + (-1)^n \big) \: ,
